can_move_capital = {
	condition = {
		tooltip = NO_CAPITAL_MOVE_SHOGUN
		potential = {
			FROM = {
				has_reform = shogunate
			}
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = NO_CAPITAL_MOVE_DUTCH_PROVINCE
		potential = {
			FROM = {
				is_subject_of_type = dutch_province_subject
			}
		}
		allow = {
			region = low_countries_region
		}
	}
}
