# Can create an Autonomous Dominion in Indian Trade Company Regions
can_not_create_autonomous_dominion_in_west_india = {
	potential = {
		NOT = { has_dlc = "Wealth of Nations" }
		NOT = { has_dlc = "Dharma" }
		is_subject = no
		# can_be_overlord = autonomous_dominion
		NOT = { is_autonomous_dominion_tag = yes }
		any_owned_province = {
			is_node_in_trade_company_region = yes
		}
		capital_scope = {
			is_node_in_trade_company_region = no
			NOT = { continent = asia }
			NOT = { continent = oceania }
		}
		NOT = {
			OR = {
				AND = {
					OR = {
						culture_group = british
						tag = GBR
						tag = ENG
					}
					NOT = { exists = DI1 }
				}
				AND = {
					OR = {
						culture_group = germanic

						culture_group = iberian
						tag = SPA

						tag = CAS
						tag = ARA

				}
				AND = {
					OR = {
						culture_group = french
						tag = FRA
						tag = FR2
						tag = BRI
						tag = BUR
					}
					NOT = { exists = DI4 }
				}
				AND = {
					OR = {
						tag = PAP
						religion_group = christian
					}
					NOT = { exists = DI5 }
				}
				AND = {
					religion_group = muslim
					NOT = { exists = DI6 }
				}
				NOT = { exists = DI7 }
				NOT = { exists = DI8 }
				NOT = { exists = DI9 }
			}
		}
		NOT = {
			any_subject_country = {
				OR = {
					tag = DI1
					tag = DI2
					tag = DI3
					tag = DI4
					tag = DI5
					tag = DI6
					tag = DI7
					tag = DI8
					tag = DI9
				}
			}
		}
		trade_company_west_india = {
			type = all
			OR = {
				NOT = { owned_by = ROOT }
				is_in_capital_area = no
			}
		}
		trade_company_deccan = {
			type = all
			OR = {
				NOT = { owned_by = ROOT }
				is_in_capital_area = no
			}
		}
		trade_company_east_india = {
			type = all
			OR = {
				NOT = { owned_by = ROOT }

				NOT = { owned_by = ROOT }

			}
		}
		trade_company_north_india = {
			type = all
			OR = {
				NOT = { owned_by = ROOT }
				is_in_capital_area = no
			}
		}
		OR = {
			trade_company_west_india = {
				owned_by = ROOT
			}

				owned_by = ROOT
