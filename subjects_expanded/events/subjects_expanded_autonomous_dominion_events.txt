namespace = subjects_expanded_autonomous_dominion_events

country_event = {
	id = subjects_expanded_autonomous_dominion_events.1
	title = subjects_expanded_autonomous_dominion_events.EVTNAME1
	desc = subjects_expanded_autonomous_dominion_events.EVTDESC1
	picture = DIPLOMACY_eventPicture
	
	hidden = yes
	is_triggered_only = yes
	
	trigger = {
		is_subject_of_type = autonomous_dominion
		OR = {
			NOT = { has_reform = colonial_government }
			NOT = { has_country_flag = was_autonomous_dominion_subject }
		}
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					NOT = { has_reform = colonial_government }
				}
				change_government = republic
				add_government_reform = colonial_government
			}
			if = {
				limit = {
					NOT = { has_country_flag = was_autonomous_dominion_subject }
				}
				set_country_flag = was_autonomous_dominion_subject
			}
		}
	}
	
	option = {
		name = subjects_expanded_autonomous_dominion_events1.EVTOPT1
		
		ai_chance = { factor = 100 }
	}
}
